# Data wrangling ---------------

# Treatment group includes only those who received treatment 
# between 2nd and 3rd waves, for simplicity.
analysis_data <- data %>%
  filter(wave >= 2) %>%
  group_by(sheno) %>%
  mutate(
      treatment = max((wave == 3) & (timetreat == 1)),
      control = all(control == 1)
    ) %>%
  filter(treatment == TRUE | control == TRUE) %>%
  ungroup() %>%
  mutate(
    treatment_group = ifelse(cashtreat == 1, "Cash",
                               ifelse(equiptreat == 1, "In-kind", "Control"))
    ) 
# Using ifelse (nested in this case) function to make the treatment_group variable that passes a given test
# Here, Cash" is assigned if the unit of observation is equal to 1 or "In Kind" if equiptreat is equal to 1
# and Control otherwise

# Create a graph ---------------
ggplot(analysis_data, aes(x = realfinalprofit, 
                          y = fct_rev(factor(wave)), 
                          color = factor(treatment_group), 
                          fill = factor(treatment_group))) +
  geom_density_ridges(alpha = 0.1, scale = 1) +
  theme_ridges() +
  labs(
    x = "3-month Real Profit (cedi)",
    y = "Rounds"
  ) +
  scale_color_brewer(palette = "Set2", name = "Group", breaks = c("Control", "Cash", "In-kind")) +
  scale_fill_brewer(palette = "Set2", name = "Group", breaks = c("Control", "Cash", "In-kind")) +
  xlim(c(0, 250))
